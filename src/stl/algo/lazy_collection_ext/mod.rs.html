4use crate::algo::collection_ext::CollectionExt;
5use crate::collections::LazyMappedCollection;
6use crate::iterators::LazyCollectionIter;
7use crate::{BidirectionalCollection, LazyCollection};

9/// Algorithms for `LazyCollection`.
10pub trait LazyCollectionExt: LazyCollection
11where
12    Self::Whole: LazyCollection,
13{
14    /// Returns the "lazily computed" first element, or nil if `self` is empty.
15    fn lazy_first(&self) -> Option<Self::Element> {
16        if self.start() == self.end() {
17            None
18        } else {
19            Some(self.compute_at(&self.start()))
20        }
21    }
22
23    /*-----------------Iteration Algorithms-----------------*/
24
25    /// Returns an iterator to iterate over lazyily computed elements in collection.
26    fn lazy_iter(&self) -> LazyCollectionIter<'_, Self::Whole> {
27        LazyCollectionIter::new(self.full())
28    }
29
30    /// Applies f to each "lazily computed" element of collection.
31    ///
32    /// # Complexity
33    ///   - O(n) where `n == self.count()`.
34    ///
35    /// # Example
36    /// ```rust
37    /// use stl::*;
38    ///
39    /// let arr = 1..4;
40    /// let mut sum = 0;
41    /// arr.lazy_for_each(|x| sum = sum + x);
42    /// assert_eq!(sum, 6);
43    /// ```
44    fn lazy_for_each<F>(&self, mut f: F)
45    where
46        F: FnMut(Self::Element),
47    {
48        let mut start = self.start();
49        let end = self.end();
50        while start != end {
51            f(self.compute_at(&start));
52            start = self.next(start);
53        }
54    }
55
56    /*-----------------Transformation algorithms-----------------*/
57
58    /// Returns a lazy collection projecting elements of mapping the given closure over lazily
59    /// computed values of self.
60    ///
61    /// # Example
62    /// ```rust
63    /// use stl::*;
64    ///
65    /// let arr = (1..=5).lazy_map(|x| x * 2);
66    /// assert!(arr.equals(&[2, 4, 6, 8, 10]));
67    /// ```
68    fn lazy_map<MapFn, MappedType>(
69        self,
70        map_fn: MapFn,
71    ) -> LazyMappedCollection<Self, MapFn, MappedType>
72    where
73        Self: Sized,
74        MapFn: Fn(Self::Element) -> MappedType,
75    {
76        LazyMappedCollection::new(self, map_fn)
77    }
78
79    /*-----------------Partition Algorithms-----------------*/
80
81    /// Returns two Vec containing the elements of the collection that
82    /// donâ€™t and do satisfy the given predicate, respectively.
83    ///
84    /// # Postcondition
85    ///   - Returns `(falseVec, trueVec)` where `falseVec` contains all elements
86    ///     that don't satisfy predicate and `trueVec` contains all elements
87    ///     that do satisfy predicate.
88    ///   - Relative ordering of elements is preserved in both Vec.
89    ///
90    /// # Complexity
91    ///   - O(n) where `n == self.count()`.
92    ///
93    /// # Examples
94    /// ```rust
95    /// use stl::*;
96    ///
97    /// let arr = 1..=5;
98    /// let (evens, odds) = arr.lazy_partitioned(|x| x % 2 == 1);
99    /// assert_eq!(evens, [2, 4]);
100    /// assert_eq!(odds, [1, 3, 5]);
101    /// ```
102    fn lazy_partitioned<F>(
103        &self,
104        mut belongs_in_second_half: F,
105    ) -> (Vec<Self::Element>, Vec<Self::Element>)
106    where
107        F: FnMut(&Self::Element) -> bool,
108    {
109        let n = self.count();
110        let mut left = Vec::with_capacity(n);
111        let arr = left.spare_capacity_mut();
112        let mut left_idx = 0;
113        let mut right_idx = n;
114
115        for e in self.lazy_iter() {
116            if belongs_in_second_half(&e) {
117                right_idx -= 1;
118                arr[right_idx].write(e);
119            } else {
120                arr[left_idx].write(e);
121                left_idx += 1;
122            }
123        }
124
125        unsafe {
126            left.set_len(n);
127        }
128
129        let mut right = left.split_off(right_idx);
130        right.reverse();
131
132        (left, right)
133    }
134
135    /*-----------------Numeric Algorithms-----------------*/
136
137    /// Returns the result of combining elements of given collection using given
138    /// accumulation operation from left to right.
139    ///
140    /// # Postcondition
141    ///   - Result is `(((init + e1) + e2) + ... + en)`.
142    ///     where e1, e2, ..., en are the lazily computed values of collection,
143    ///     where (a + b) represents op(a, b).
144    ///
145    /// # Complexity:
146    ///   - O(`count`)
147    ///
148    /// # Examples
149    /// ```rust
150    /// use stl::*;
151    ///
152    /// let arr = 1..=3;
153    /// assert_eq!(arr.lazy_fold_left(0, |x, y| x + y), 6);
154    /// ```
155    fn lazy_fold_left<R, F>(&self, init: R, mut op: F) -> R
156    where
157        F: FnMut(R, Self::Element) -> R,
158    {
159        let mut res = init;
160        for e in self.lazy_iter() {
161            res = op(res, e)
162        }
163        res
164    }
165
166    /// Returns the result of combining elements of given collection using given
167    /// accumulation operation from right to left.
168    ///
169    /// # Postcondition
170    ///   - Result is `(e0 + ... + (e(n-1) + (e(n) + init)))`.
171    ///     where e1, e2, ..., en are the references to collection elements,
172    ///     where (a + b) represents op(a, b).
173    ///
174    /// # Complexity:
175    ///   - O(`count`)
176    ///
177    /// # Examples
178    /// ```rust
179    /// use stl::*;
180    ///
181    /// let arr = 1..=3;
182    /// assert_eq!(arr.lazy_fold_right(0, |x, y| x + y), 6);
183    /// ```
184    fn lazy_fold_right<R, F>(&self, init: R, mut op: F) -> R
185    where
186        F: FnMut(Self::Element, R) -> R,
187        Self: BidirectionalCollection,
188        Self::Whole: BidirectionalCollection,
189    {
190        let mut res = init;
191        let mut rest = self.full();
192        while let Some(e) = rest.lazy_pop_last() {
193            res = op(e, res)
194        }
195        res
196    }
197}

200where
201    R: LazyCollection + ?Sized,
202    R::Whole: LazyCollection,
203{
