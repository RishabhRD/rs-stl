4use crate::{Collection, LazyCollection, LazyCollectionIterator};
5
6/// Algorithms for `LazyCollection`.
7pub trait LazyCollectionExt: LazyCollection
8where
9    Self::Whole: LazyCollection,
10{
11    /// Returns the "lazily computed" first element, or nil if `self` is empty.
12    fn lazy_first(&self) -> Option<<Self as Collection>::Element> {
13        if self.start() == self.end() {
14            None
15        } else {
16            Some(self.compute_at(&self.start()))
17        }
18    }
19
20    /// Applies f to each "lazily computed" element of collection.
21    ///
22    /// # Precondition
23    ///
24    /// # Postcondition
25    ///   - Applies f to each element of collection.
26    ///   - Complexity: O(n). Exactly n applications of f.
27    ///
28    /// # Example
29    /// ```rust
30    /// use stl::*;
31    ///
32    /// let arr = 1..4;
33    /// let mut sum = 0;
34    /// arr.lazy_for_each(|x| sum = sum + x);
35    /// assert_eq!(sum, 6);
36    /// ```
37    fn lazy_for_each<F>(&self, mut f: F)
38    where
39        F: FnMut(Self::Element),
40    {
41        let mut start = self.start();
42        let end = self.end();
43        while start != end {
44            f(self.compute_at(&start));
45            start = self.next(start);
46        }
47    }
48
49    /// Returns a non-consuming iterator that iterates over `Self::Element`.
50    /// The iterator lazily computes the value on each `next` call.
51    fn lazy_iter(&self) -> LazyCollectionIterator<Self::Whole> {
52        LazyCollectionIterator::new(self.slice(self.start(), self.end()))
53    }
54}
55
56impl<R> LazyCollectionExt for R
57where
58    R: LazyCollection + ?Sized,
59    R::Whole: LazyCollection,
60{
61}
