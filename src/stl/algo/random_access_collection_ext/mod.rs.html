4use crate::{RandomAccessCollection, ReorderableCollection};
5mod sort;
6
7/// Algorithms for `RandomAccessCollection`.
8pub trait RandomAccessCollectionExt: RandomAccessCollection
9where
10    Self::Whole: RandomAccessCollection,
11{
12    /*-----------------Sorting Algorithms-----------------*/
13
14    /// Sorts the collection in place, using the given predicate as comparision between elements.
15    ///
16    /// # Precondition:
17    ///   - `are_in_increasing_order` should follow strict weak ordering.
18    ///
19    /// # Postcondition:
20    ///   - Relative ordering of equivalent elements are NOT guaranteed to be presevered.
21    ///
22    /// # Complexity:
23    ///   - O(n * log(n)) worst case where `n == self.count()`.
24    ///
25    /// # Example
26    /// ```rust
27    /// use stl::*;
28    ///
29    /// let mut arr = [3, 4, 1, 2, 5];
30    /// arr.sort_unstable_by(|x, y| x < y);
31    /// assert_eq!(arr, [1, 2, 3, 4, 5]);
32    /// ```
33    fn sort_unstable_by<Compare>(&mut self, are_in_increasing_order: Compare)
34    where
35        Self: ReorderableCollection,
36        Self::Whole: ReorderableCollection,
37        Compare: Fn(&Self::Element, &Self::Element) -> bool + Clone,
38    {
39        sort::sort_unstable_by(self, are_in_increasing_order);
40    }
41
42    /// Sorts the collection in place.
43    ///
44    /// # Postcondition:
45    ///   - Relative ordering of equivalent elements are NOT guaranteed to be presevered.
46    ///
47    /// # Complexity:
48    ///   - O(n * log(n)) worst case where `n == self.count()`.
49    ///
50    /// # Example
51    /// ```rust
52    /// use stl::*;
53    ///
54    /// let mut arr = [3, 4, 1, 2, 5];
55    /// arr.sort_unstable();
56    /// assert_eq!(arr, [1, 2, 3, 4, 5]);
57    /// ```
58    fn sort_unstable(&mut self)
59    where
60        Self: ReorderableCollection,
61        Self::Whole: ReorderableCollection,
62        Self::Element: Ord,
63    {
64        self.sort_unstable_by(|x, y| x < y)
65    }
66}
67
68impl<R> RandomAccessCollectionExt for R
69where
70    R: RandomAccessCollection + ?Sized,
71    R::Whole: RandomAccessCollection,
72{
