10/// Algorithms for `Collection`.
11pub trait CollectionExt: Collection {
12    /// Returns true if given collection is empty.
13    ///
14    /// # Complexity
15    ///   - O(1).
16    fn is_empty(&self) -> bool {
17        self.start() == self.end()
18    }
19
20    /*-----------------Element Access Algorithms-----------------*/
21
22    /// Returns the first element, or nil if `self` is empty.
23    fn first(&self) -> Option<Self::ElementRef<'_>> {
24        if self.start() == self.end() {
25            None
26        } else {
27            Some(self.at(&self.start()))
28        }
29    }
30
31    /*-----------------Slice Algorithms-----------------*/
32
33    /// Returns slice of the collection covering full collection.
34    ///
35    /// # Complexity
36    ///   - O(1).
37    ///
38    /// # Examples
39    /// ```rust
40    /// use stl::*;
41    ///
42    /// let arr = [1, 2, 3, 4, 5];
43    /// let s = arr.full();
44    /// assert!(s.equals(&[1, 2, 3, 4, 5]));
45    /// ```
46    fn full(&self) -> Slice<'_, Self::Whole> {
47        self.slice(self.start(), self.end())
48    }
49
50    /// Returns a slice, upto specified maximum length, containing the initial elements of
51    /// collection.
52    ///
53    /// # Postcondition
54    ///   - If the maximum length exceeds the number of elements in the
55    ///     collection, the result contains all elements in the collection.
56    ///
57    /// # Complexity
58    ///   - O(1) for RandomAccessCollection; otherwise O(k) where k is the
59    ///     number of elements in resultant slice.
60    ///
61    /// # Example
62    /// ```rust
63    /// use stl::*;
64    ///
65    /// let arr = [1, 2, 3, 4, 5];
66    /// let s = arr.prefix(3);
67    /// assert!(s.equals(&[1, 2, 3]));
68    /// ```
69    fn prefix(&self, max_length: usize) -> Slice<'_, Self::Whole> {
70        let mut end = self.start();
71        self.form_next_n_limited_by(&mut end, max_length, self.end());
72        self.prefix_upto(end)
73    }
74
75    /// Returns prefix slice of the collection ending at `to` exclusive.
76    ///
77    /// # Precondition
78    ///   - `to` is a valid position in the collection.
79    ///
80    /// # Complexity
81    ///   - O(1).
82    ///
83    /// # Examples
84    /// ```rust
85    /// use stl::*;
86    ///
87    /// let arr = [1, 2, 3, 4, 5];
88    /// let p = arr.prefix_upto(3);
89    /// assert!(p.equals(&[1, 2, 3]));
90    /// ```
91    fn prefix_upto(&self, pos: Self::Position) -> Slice<'_, Self::Whole> {
92        self.slice(self.start(), pos)
93    }
94
95    /// Returns prefix slice of the collection ending at `to` inclusive.
96    ///
97    /// # Precondition
98    ///   - `to` is a valid position in the collection and `to != self.end()`.
99    ///
100    /// # Complexity
101    ///   - O(1).
102    ///
103    /// # Examples
104    /// ```rust
105    /// use stl::*;
106    ///
107    /// let arr = [1, 2, 3, 4, 5];
108    /// let p = arr.prefix_through(3);
109    /// assert!(p.equals(&[1, 2, 3, 4]));
110    /// ```
111    fn prefix_through(&self, pos: Self::Position) -> Slice<'_, Self::Whole> {
112        self.prefix_upto(self.next(pos))
113    }
114
115    /// Returns prefix slice containing the initial elements until `predicate` returns false and
116    /// skipping the remaining elements.
117    ///
118    /// # Complexity
119    ///   - O(n) where `n == self.count()`.
120    ///
121    /// # Examples
122    /// ```rust
123    /// use stl::*;
124    ///
125    /// let arr = [1, 3, 5, 2, 7];
126    /// let p = arr.prefix_while(|x| x % 2 == 1);
127    /// assert!(p.equals(&[1, 3, 5]));
128    /// ```
129    fn prefix_while<F: FnMut(&Self::Element) -> bool>(
130        &self,
131        mut predicate: F,
132    ) -> Slice<'_, Self::Whole> {
133        let p = self
134            .first_position_where(|x| !predicate(x))
135            .unwrap_or(self.end());
136        self.prefix_upto(p)
137    }
138
139    /// Returns a slice by skipping initial elements while `predicate` returns true,
140    /// returning the remaining elements.
141    ///
142    /// # Complexity
143    ///   - O(n) where `n == self.count()`.
144    ///
145    /// # Examples
146    /// ```rust
147    /// use stl::*;
148    ///
149    /// let arr = [1, 3, 5, 2, 4, 7];
150    /// let s = arr.dropping_while(|x| x % 2 == 1);
151    /// assert!(s.equals(&[2, 4, 7]));
152    /// ```
153    fn dropping_while<F>(&self, mut predicate: F) -> Slice<'_, Self::Whole>
154    where
155        F: FnMut(&Self::Element) -> bool,
156    {
157        self.suffix_from(
158            self.first_position_where(|x| !predicate(x))
159                .unwrap_or(self.end()),
160        )
161    }
162
163    /// Returns a slice containing all but the given number of initial elements.
164    ///
165    /// # Postcondition
166    ///   - If `count > self.count()`, returns an empty slice.
167    ///
168    /// # Complexity
169    ///   - O(1) for RandomAccessCollection;
170    ///   - O(k) otherwise, where k is number of elements to drop from beginning of collection.
171    ///
172    /// # Examples
173    /// ```rust
174    /// use stl::*;
175    ///
176    /// let arr = [1, 2, 3, 4, 5];
177    /// let s = arr.dropping_prefix(3);
178    /// assert!(s.equals(&[4, 5]));
179    /// ```
180    fn dropping_prefix(&self, count: usize) -> Slice<'_, Self::Whole> {
181        let mut start = self.start();
182        self.form_next_n_limited_by(&mut start, count, self.end());
183        self.suffix_from(start)
184    }
185
186    /// Returns a slice containing all but the given number of final elements.
187    ///
188    /// # Postcondition
189    ///   - If `count > self.count()`, returns an empty slice.
190    ///
191    /// # Complexity
192    ///   - O(1) for RandomAccessCollection;
193    ///   - O(n) otherwise, where `n == self.count()`.
194    ///
195    /// # Examples
196    /// ```rust
197    /// use stl::*;
198    ///
199    /// let arr = [1, 2, 3, 4, 5];
200    /// let s = arr.dropping_suffix(3);
201    /// assert!(s.equals(&[1, 2]));
202    /// ```
203    fn dropping_suffix(&self, count: usize) -> Slice<'_, Self::Whole> {
204        let n = self.count();
205        if count > n {
206            return self.prefix_upto(self.start());
207        }
208        self.prefix_upto(self.next_n(self.start(), n - count))
209    }

211    /// Returns a slice, upto specified maximum length, containing the final elements of the
212    /// collection.
213    ///
214    /// # Postcondition
215    ///   - If the maximum length exceeds the number of elements in the
216    ///     collection, the result contains all elements in the collection.
217    ///
218    /// # Complexity
219    ///   - O(1) for RandomAccessCollection; otherwise O(n) where
220    ///     `n == self.count()`.
221    ///
222    /// # Example
223    /// ```rust
224    /// use stl::*;
225    ///
226    /// let arr = [1, 2, 3, 4, 5];
227    /// let s = arr.suffix(3);
228    /// assert!(s.equals(&[3, 4, 5]));
229    /// ```
230    fn suffix(&self, max_length: usize) -> Slice<'_, Self::Whole> {
231        let n = self.count();
232        if max_length > n {
233            self.full()
234        } else {
235            self.suffix_from(self.next_n(self.start(), n - max_length))
236        }
237    }

239    /// Returns suffix slice of the collection starting from `from` inclusive.
240    ///
241    /// # Precondition
242    ///   - `from` is a valid position in the collection.
243    ///
244    /// # Complexity
245    ///   - O(1).
246    ///
247    /// # Examples
248    /// ```rust
249    /// use stl::*;
250    ///
251    /// let arr = [1, 2, 3, 4, 5];
252    /// let s = arr.suffix_from(3);
253    /// assert!(s.equals(&[4, 5]));
254    /// ```
255    fn suffix_from(&self, from: Self::Position) -> Slice<'_, Self::Whole> {
256        self.slice(from, self.end())
257    }

259    /// Returns two disjoint slices of `self` split at the given `position`.
260    ///
261    /// # Examples
262    /// ```rust
263    /// use stl::*;
264    ///
265    /// let arr = [0, 1, 2, 3, 4];
266    /// let (s1, s2) = arr.splitting_at(2);
267    /// assert!(s1.equals(&[0, 1]));
268    /// assert!(s2.equals(&[2, 3, 4]));
269    /// ```
270    fn splitting_at(
271        &self,
272        position: Self::Position,
273    ) -> (Slice<'_, Self::Whole>, Slice<'_, Self::Whole>) {
274        self.full().split_at(position)
275    }

277    /// Returns two disjoint slices of `self`, split immediately *after* the
278    /// given `position`.
279    ///
280    /// # Precondition
281    ///   - `position != self.end()`.
282    ///
283    /// # Examples
284    /// ```rust
285    /// use stl::*;
286    ///
287    /// let arr = [0, 1, 2, 3, 4];
288    /// let (s1, s2) = arr.splitting_after(2);
289    /// assert!(s1.equals(&[0, 1, 2]));
290    /// assert!(s2.equals(&[3, 4]));
291    /// ```
292    fn splitting_after(
293        &self,
294        position: Self::Position,

297    }

302    fn iter(&self) -> CollectionIter<'_, Self::Whole> {

304    }

306    /// Applies f to each element of collection.
307    ///
308    /// # Complexity:
309    ///   - O(n) where `n == self.count()`.
310    ///
311    /// # Example
312    /// ```rust
313    /// use stl::*;
314    ///
315    /// let arr = [1, 2, 3];
316    /// let mut sum = 0;
317    /// arr.for_each(|x| sum = sum + x);
318    /// assert_eq!(sum, 6);
319    /// ```
320    fn for_each<F>(&self, mut f: F)
321    where
322        F: FnMut(&Self::Element),
323    {
324        let mut start = self.start();
325        let end = self.end();
326        while start != end {
327            f(&self.at(&start));
328            start = self.next(start);
329        }
330    }

332    /// Returns an iterator of slices which are separated by elements that match `pred`.

344    /// assert_eq!(res, vec![9, 0, 3, 10]);

348        pred: Pred,
349    ) -> SplitWhereIterator<'_, Self::Whole, Pred>

362    ///   - `max_slices > 0`,

448    /// Returns true if elements of self is equivalent to elements of other by given relation bi_pred.
449    ///
450    /// # Postcondition
451    ///   - Returns true if elements of self is equivalent to other by given relation bi_pred.
452    ///   - If self and other have different number of elements, then return false.
453    ///
454    /// # Complexity
455    ///   - `O(min(m, n))`
456    ///     where
457    ///     - `m == self.count()`
458    ///     - `n == other.count()`
459    ///
460    /// # Examples
461    /// ```rust
462    /// use stl::*;
463    ///
464    /// let arr1 = [1, 2, 3];
465    /// let arr2 = [2, 3, 4];
466    /// assert!(arr1.equals_by(&arr2, |x, y| *y == x + 1));
467    /// ```
468    fn equals_by<OtherCollection, F>(
469        &self,
470        other: &OtherCollection,
471        mut bi_pred: F,
472    ) -> bool
473    where
474        OtherCollection: Collection,
475        F: FnMut(&Self::Element, &OtherCollection::Element) -> bool,
476    {
477        let mut self1 = self.full();
478        let mut other1 = other.full();
479        loop {
480            match (self1.pop_first(), other1.pop_first()) {
481                (Some(x), Some(y)) if bi_pred(&x, &y) => {}
482                (None, None) => return true,
483                _ => return false,
484            }
485        }
486    }

489    ///
490    /// # Postcondition
491    ///   - Returns true if elements of self is equal to elements of other.
492    ///   - If self and other have different number of elements, then return false.
493    ///
494    /// # Complexity
495    ///   - `O(min(m, n))`
496    ///     where
497    ///     - `m == self.count()`
498    ///     - `n == other.count()`
499    ///
500    /// # Examples
501    /// ```rust
502    /// use stl::*;
503    ///
504    /// let arr1 = [1, 2, 3];
505    /// let arr2 = [1, 2, 3];
506    /// assert!(arr1.equals(&arr2));

510        OtherCollection: Collection<Element = Self::Element>,

518    /// Finds position of first element in `self` satisfying `pred`. If no such
519    /// element exists, returns `self.end()`.
520    ///
521    /// # Complexity
522    ///   - O(n) where `n == self.count()`.
523    ///
524    /// # Example
525    /// ```rust
526    /// use stl::*;
527    ///
528    /// let arr = [1, 2, 3];
529    /// let i = arr.first_position_where(|x| *x == 3);
530    /// assert_eq!(i, Some(2));
531    /// ```
532    fn first_position_where<Pred>(
533        &self,
534        mut pred: Pred,
535    ) -> Option<Self::Position>
536    where
537        Pred: FnMut(&Self::Element) -> bool,
538    {
539        let mut rest = self.full();
540        let mut p = self.start();
541        while let Some(e) = rest.pop_first() {
542            if pred(&e) {
543                return Some(p);
544            }
545            p = rest.start()
546        }
547        None
548    }

735    ///
736    /// # Complexity
737    ///   - O(n) where `n == self.count()`.
738    ///
739    /// # Example
740    /// ```rust
741    /// use stl::*;
742    ///
743    /// let arr = [3, 0, 3];
744    /// let n = arr.count_of(&3);
745    /// assert_eq!(n, 2);
746    /// ```
747    fn count_of(&self, e: &Self::Element) -> usize
748    where
749        Self::Element: Eq,
750    {

756    /// Returns position of first element of collection for which predicate returns false.
757    ///
758    /// # Precondition
759    ///   - The collection should be already partitioned wrt predicate i.e,
760    ///     there exist a position `i` such that predicate is true for every
761    ///     element of `self.prefix_upto(i)` and predicate is false for every
762    ///     element of `self.suffix_from(i)`.
763    ///
764    /// # Complexity
765    ///   - O(log n) for RandomAccessCollection, O(n) otherwise; where `n == self.count()`.
766    ///   - O(log n) application of `belongs_in_second_half`.
767    ///
768    /// # Example
769    /// ```rust
770    /// use stl::*;
771    ///
772    /// let arr = [1, 3, 5, 2, 4];
773    /// let i = arr.partition_point(|x| x % 2 == 0);
774    /// assert_eq!(i, 3);
775    /// ```
776    fn partition_point<F>(
777        &self,
778        mut belongs_in_second_half: F,
779    ) -> Self::Position
780    where
781        F: FnMut(&Self::Element) -> bool,
782    {
783        let mut f = self.start();
784        let mut n = self.count();
785        while n > 0 {
786            let half = n / 2;
787            let m = self.next_n(f.clone(), half);
788            if belongs_in_second_half(&self.at(&m)) {
789                n = half;
790            } else {
791                f = self.next(m);
792                n -= half + 1;
793            }
794        }
795        f
796    }

798    /// Returns two Vec containing the elements of the collection that
799    /// dont and do satisfy the given predicate, respectively.
800    ///
801    /// # Postcondition
802    ///   - Returns `(falseVec, trueVec)` where `falseVec` contains all elements
803    ///     that don't satisfy predicate and `trueVec` contains all elements
804    ///     that do satisfy predicate.
805    ///   - Relative ordering of elements is preserved in both Vec.
806    ///
807    /// # Complexity
808    ///   - O(n) where `n == self.count()`.
809    ///
810    /// # Examples
811    /// ```rust
812    /// use stl::*;
813    ///
814    /// let arr = [1, 2, 3, 4, 5];
815    /// let (evens, odds) = arr.partitioned(|x| x % 2 == 1);
816    /// assert_eq!(evens, [2, 4]);
817    /// assert_eq!(odds, [1, 3, 5]);
818    /// ```
819    fn partitioned<F>(
820        &self,
821        mut belongs_in_second_half: F,
822    ) -> (Vec<Self::Element>, Vec<Self::Element>)
823    where
824        Self::Element: Clone,
825        F: FnMut(&Self::Element) -> bool,
826    {
827        let n = self.count();
828        let mut left = Vec::with_capacity(n);
829        let arr = left.spare_capacity_mut();
830        let mut left_idx = 0;
831        let mut right_idx = n;
832
833        let mut rest = self.full();
834        while let Some(e) = rest.pop_first() {
835            if belongs_in_second_half(&e) {
836                right_idx -= 1;
837                arr[right_idx].write((*e).clone());
838            } else {
839                arr[left_idx].write((*e).clone());
840                left_idx += 1;
841            }
842        }
843
844        unsafe {
845            left.set_len(n);
846        }
847
848        let mut right = left.split_off(right_idx);
849        right.reverse();
850
851        (left, right)
852    }

856    /// Returns the result of combining elements of given collection using given
857    /// accumulation operation from left to right.
858    ///
859    /// # Postcondition
860    ///   - Result is `(((init + e1) + e2) + ... + en)`.
861    ///     where e1, e2, ..., en are the references to collection elements,
862    ///     where (a + b) represents op(a, b).
863    ///
864    /// # Complexity:
865    ///   - O(`count`)
866    ///
867    /// # Examples
868    /// ```rust
869    /// use stl::*;
870    ///
871    /// let arr = [1, 2, 3];
872    /// assert_eq!(arr.fold_left(0, |x, y| x + y), 6);
873    /// ```
874    fn fold_left<R, F>(&self, init: R, mut op: F) -> R
875    where
876        F: FnMut(R, &Self::Element) -> R,
877    {
878        let mut res = init;
879        let mut rest = self.full();
880        while let Some(e) = rest.pop_first() {
881            res = op(res, &e)
882        }
883        res
884    }

888    /// Copies and returns all elements of `self` into `Vec<Self::Element>`.
889    ///
890    /// # Complexity
891    ///   - O(`self.count()`)
892    fn to_vec(&self) -> Vec<Self::Element>
893    where
894        Self::Element: Clone,
895    {
896        let mut r = Vec::with_capacity(self.underestimated_count());
897        for e in self.iter() {
898            r.push(e.clone())
899        }
900        r
901    }
